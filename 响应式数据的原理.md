## 响应式数据的原理



```js
let obj = {
  name: 'lys',
  age: 20,
  wife: {
    name: 'wjx',
    age: 20
  }
}
let p = new Proxy(obj, {
    get(target, prop) {
        console.log('target', target, prop);
        return target[prop];
    },
    set (target, prop, val) {
			console.log('set');
      target[prop] = val;
    },
  deleteProperty (target, prop) {
    console.log('delete', target, prop);
    // delete target[prop];
    return Reflect.deleteProperty(target, prop);
  }
  
});

console.log(p.name);
delete p.name;


```
